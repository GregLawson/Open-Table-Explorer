def single_grep(context, pattern)
	regexp=Regexp.new(pattern)
	matchData=regexp.match(self)
	if matchData then
		ActiveSupport::HashWithIndifferentAccess.new(:context => context, :matchData => matchData)
	else
		nil #don't select line for return
	end #if
end
